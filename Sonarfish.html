// AI-Reel X1 - Fishing Reel Control System
// github.com/swordenkisk

#include <Wire.h>
#include <Servo.h>
#include <MPU6050.h>

// Pin Definitions
const int MOTOR_PWM = 9;
const int MOTOR_DIR = 8;
const int LOAD_CELL_DOUT = 3;
const int LOAD_CELL_SCK = 2;
const int CAST_SOLENOID = 7;
const int EMERGENCY_STOP = 4;

// Constants
const float MAX_DRAG_KG = 150.0;
const float BITE_THRESHOLD = 2.5;  // Newtons
const int SAMPLE_RATE = 1000;        // Hz

// State Variables
float currentTension = 0.0;
float estimatedWeight = 0.0;
bool isFishing = false;
bool biteDetected = false;

MPU6050 mpu;

void setup() {
    Serial.begin(115200);
    Wire.begin();
    mpu.initialize();
    
    pinMode(MOTOR_PWM, OUTPUT);
    pinMode(MOTOR_DIR, OUTPUT);
    pinMode(CAST_SOLENOID, OUTPUT);
    pinMode(EMERGENCY_STOP, INPUT_PULLUP);
    
    Serial.println("AI-Reel X1 Initialized");
}

void loop() {
    // Emergency stop check
    if (digitalRead(EMERGENCY_STOP) == LOW) {
        emergencyStop();
        return;
    }
    
    // Read sensors at 1000Hz
    currentTension = readLoadCell();
    int16_t ax, ay, az;
    mpu.getAcceleration(&ax, &ay, &az);
    
    // AI Bite Detection Algorithm
    if (!biteDetected && detectBite(currentTension, ax, ay, az)) {
        biteDetected = true;
        estimatedWeight = estimateWeight(currentTension);
        Serial.print("Bite detected! Est. weight: ");
        Serial.print(estimatedWeight);
        Serial.println(" kg");
        activateAutoReel();
    }
    
    // Auto-reel logic
    if (biteDetected && isFishing) {
        adaptiveReelControl(currentTension, estimatedWeight);
    }
    
    delayMicroseconds(1000);  // 1ms = 1000Hz
}

bool detectBite(float tension, int16_t ax, int16_t ay, int16_t az) {
    // Multi-sensor fusion for bite detection
    float vibration = sqrt(ax*ax + ay*ay + az*az) / 16384.0;
    bool tensionSpike = tension > BITE_THRESHOLD;
    bool patternMatch = analyzeVibrationPattern(vibration);
    
    return tensionSpike && patternMatch;
}

float estimateWeight(float initialTension) {
    // Weight estimation from initial pull force
    return (initialTension * 0.8) / 9.81;  // Rough estimate in kg
}

void activateAutoReel() {
    isFishing = true;
    // Start with moderate drag
    analogWrite(MOTOR_PWM, 100);
    digitalWrite(MOTOR_DIR, HIGH);
}

void adaptiveReelControl(float tension, float weight) {
    // Adaptive control based on fish behavior
    int pwmValue;
    
    if (weight > 50.0) {
        // Heavy fish: slow and steady
        pwmValue = map(tension, 0, MAX_DRAG_KG * 9.81, 80, 180);
    } else {
        // Light fish: faster retrieve
        pwmValue = map(tension, 0, MAX_DRAG_KG * 9.81, 120, 255);
    }
    
    // Safety: cut if approaching line limit
    if (tension > MAX_DRAG_KG * 9.81 * 0.9) {
        emergencyStop();
        return;
    }
    
    analogWrite(MOTOR_PWM, pwmValue);
}

void emergencyStop() {
    analogWrite(MOTOR_PWM, 0);
    digitalWrite(CAST_SOLENOID, LOW);
    isFishing = false;
    biteDetected = false;
    Serial.println("EMERGENCY STOP ACTIVATED");
}

void autoCast() {
    // One-button cast mechanism
    digitalWrite(CAST_SOLENOID, HIGH);
    delay(200);
    digitalWrite(CAST_SOLENOID, LOW);
    Serial.println("Cast deployed");
}

float readLoadCell() {
    // HX711 load cell reading
    return 0.0;  // Placeholder
}

bool analyzeVibrationPattern(float vibration) {
    // AI pattern matching placeholder
    return vibration > 1.2 && vibration < 3.5;
}
